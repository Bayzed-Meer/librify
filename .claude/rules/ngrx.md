# NgRx Rules

Every feature slice lives in `src/app/features/<name>/store/` with exactly 5 files.

## state.ts — interface + initialState only
```typescript
export interface BooksState {
  books: Book[];
  selectedBookId: string | null;
  loading: boolean;
  error: string | null;
}
export const initialBooksState: BooksState = {
  books: [], selectedBookId: null, loading: false, error: null,
};
```

## actions.ts — createActionGroup only, never createAction()
```typescript
export const BooksActions = createActionGroup({
  source: 'Books',
  events: {
    'Load Books': emptyProps(),
    'Load Books Success': props<{ books: Book[] }>(),
    'Load Books Failure': props<{ error: string }>(),
  },
});
```

## reducer.ts — createFeature, re-export generated selectors
```typescript
export const booksFeature = createFeature({
  name: 'books',
  reducer: createReducer(
    initialBooksState,
    on(BooksActions.loadBooks, (state) => ({ ...state, loading: true, error: null })),
    on(BooksActions.loadBooksSuccess, (state, { books }) => ({ ...state, books, loading: false })),
    on(BooksActions.loadBooksFailure, (state, { error }) => ({ ...state, error, loading: false })),
  ),
});
export const { selectBooksState, selectBooks, selectLoading, selectError } = booksFeature;
```

## effects.ts — functional effects with inject(), always { functional: true }
```typescript
export const loadBooks$ = createEffect(
  (actions$ = inject(Actions), booksService = inject(BooksService)) =>
    actions$.pipe(
      ofType(BooksActions.loadBooks),
      switchMap(() =>
        booksService.getBooks().pipe(
          map((books) => BooksActions.loadBooksSuccess({ books })),
          catchError((error: unknown) =>
            of(BooksActions.loadBooksFailure({ error: String(error) })),
          ),
        ),
      ),
    ),
  { functional: true },
);
```

## selectors.ts — DERIVED/COMPOSED selectors only (basics auto-generated by createFeature)
```typescript
export const selectSelectedBook = createSelector(
  booksFeature.selectBooks,
  booksFeature.selectSelectedBookId,
  (books, id) => books.find((b) => b.id === id) ?? null,
);
```

## Registration — route config only, never app.config.ts
```typescript
export const BOOKS_ROUTES: Routes = [{
  path: '',
  providers: [provideState(booksFeature), provideEffects([loadBooks$])],
  component: BooksComponent,
}];
```

## Store in Components — always bridge to signals via toSignal()
```typescript
private readonly store = inject(Store);
protected readonly books = toSignal(this.store.select(selectBooks), { initialValue: [] });
protected readonly loading = toSignal(this.store.select(selectLoading), { initialValue: false });
```

## Rules Summary
- Never use individual `createAction()` — always `createActionGroup`
- Never put composed selectors in reducer file — they go in `selectors.ts`
- Never register feature state in `app.config.ts` — use route `providers`
- Never mutate state in reducers — always spread to new objects
- Use `switchMap` for cancellable, `exhaustMap` for non-concurrent, `mergeMap` for fire-and-forget
